
class TableRow:
    def __init__(self, table_name: str, data: dict):
        """
        Obecný model pro řádek v libovolné tabulce.

        :param table_name: Název tabulky v databázi.
        :param data: Slovník s daty, kde klíče jsou názvy sloupců.
        """
        self.table_name = table_name
        self.data = data
